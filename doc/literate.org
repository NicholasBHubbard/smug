    (defun run-program (program &rest args
                        &aux process)
      (let* ((string (with-output-to-string (s) 
                       (setf process (sb-ext:run-program program args 
                                                         :search t :output s))))) 
        (values (string-right-trim '(#\Newline) string) process)))
